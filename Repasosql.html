1. INTRODUCCIÓN

¿Qué es PL/SQL?

PL/SQL (Procedural Language/Structured Query Language) es la extensión procedimental de Oracle para SQL. Permite crear programas estructurados con variables, constantes, procedimientos, funciones, y manejo de excepciones.

Configuración del Entorno:
  • Oracle Database 18c
  • SQL Developer
  • JDK 17
  • Habilitar DBMS_OUTPUT: SET SERVEROUTPUT ON

3. CONCEPTOS BÁSICOS

3.1 Variables Escalares

BÁSICO

Definición

Las variables escalares almacenan un solo valor de un tipo de datos específico. Son la base para el almacenamiento temporal de datos en PL/SQL.

DECLARE v_nombre VARCHAR2(50) := 'Juan Pérez'; v_salario NUMBER(8,2) := 850000.00; v_fecha_ingreso DATE := SYSDATE; v_activo BOOLEAN := TRUE; BEGIN DBMS_OUTPUT.PUT_LINE('Empleado: ' || v_nombre); DBMS_OUTPUT.PUT_LINE('Salario: $' || TO_CHAR(v_salario, '999,999,999.99')); END; /
Casos de Uso:
  • Almacenar valores temporales en cálculos
  • Parámetros de entrada en procedimientos
  • Contadores en ciclos
  • Flags de control de flujo

3.2 Bind Variables

BÁSICO

Definición

Las bind variables son variables de host que se pueden usar para pasar valores entre el entorno host y PL/SQL, mejorando el rendimiento al evitar análisis repetidos.

-- En SQL Developer VARIABLE v_empleado_id NUMBER VARIABLE v_resultado VARCHAR2(100) BEGIN :v_empleado_id := 100; :v_resultado := 'Procesado correctamente'; END; / PRINT v_empleado_id v_resultado
Casos de Uso:
  • Pasar parámetros desde aplicaciones
  • Optimizar consultas repetitivas
  • Intercambio de datos entre sesiones

3.3 Impresiones por Pantalla

BÁSICO

DBMS_OUTPUT.PUT_LINE

Procedimiento para mostrar texto en la consola durante la ejecución de PL/SQL.

BEGIN -- Habilitar salida (en SQL Developer) DBMS_OUTPUT.ENABLE(1000000); -- Imprimir línea completa DBMS_OUTPUT.PUT_LINE('Mensaje completo'); -- Imprimir sin salto de línea y luego salto DBMS_OUTPUT.PUT('Parte 1 '); DBMS_OUTPUT.PUT('Parte 2'); DBMS_OUTPUT.NEW_LINE; -- Formateo de números DBMS_OUTPUT.PUT_LINE('Precio: $' || TO_CHAR(15000, '999,999')); END; /

3.4 Bloques Anónimos Anidados

BÁSICO

Definición

Los bloques anónimos anidados permiten crear subprogramas temporales dentro de un bloque principal, proporcionando mejor organización y alcance de variables.

DECLARE v_total NUMBER := 0; v_contador NUMBER := 0; BEGIN DBMS_OUTPUT.PUT_LINE('Bloque principal iniciado'); -- Bloque anidado 1 DECLARE v_local NUMBER := 100; BEGIN v_total := v_total + v_local; DBMS_OUTPUT.PUT_LINE('Bloque anidado 1: ' || v_local); END; -- Bloque anidado 2 DECLARE v_local NUMBER := 200; BEGIN v_total := v_total + v_local; DBMS_OUTPUT.PUT_LINE('Bloque anidado 2: ' || v_local); END; DBMS_OUTPUT.PUT_LINE('Total final: ' || v_total); END; /
Casos de Uso:
  • Organizar lógica compleja
  • Crear variables con alcance limitado
  • Manejo específico de excepciones por sección

3.5 Condicionales IF

BÁSICO

Estructura IF-THEN-ELSE

Permite ejecutar código de forma condicional basado en expresiones booleanas.

DECLARE v_salario NUMBER := 800000; v_categoria VARCHAR2(20); v_bono NUMBER; BEGIN IF v_salario > 1500000 THEN v_categoria := 'EJECUTIVO'; v_bono := v_salario * 0.15; ELSIF v_salario > 1000000 THEN v_categoria := 'SUPERVISOR'; v_bono := v_salario * 0.10; ELSIF v_salario > 500000 THEN v_categoria := 'EMPLEADO'; v_bono := v_salario * 0.05; ELSE v_categoria := 'TRAINEE'; v_bono := 0; END IF; DBMS_OUTPUT.PUT_LINE('Categoría: ' || v_categoria); DBMS_OUTPUT.PUT_LINE('Bono: $' || TO_CHAR(v_bono, '999,999')); END; /

3.6 Condicionales CASE

BÁSICO

Estructura CASE

Alternativa más elegante a múltiples IF-ELSIF cuando se evalúa una expresión contra múltiples valores.

DECLARE v_dia_semana NUMBER := TO_CHAR(SYSDATE, 'D'); v_nombre_dia VARCHAR2(20); v_tipo_dia VARCHAR2(20); BEGIN -- CASE simple v_nombre_dia := CASE v_dia_semana WHEN 1 THEN 'Domingo' WHEN 2 THEN 'Lunes' WHEN 3 THEN 'Martes' WHEN 4 THEN 'Miércoles' WHEN 5 THEN 'Jueves' WHEN 6 THEN 'Viernes' WHEN 7 THEN 'Sábado' ELSE 'Desconocido' END; -- CASE con condiciones v_tipo_dia := CASE WHEN v_dia_semana IN (1, 7) THEN 'Fin de semana' WHEN v_dia_semana IN (2, 3, 4, 5, 6) THEN 'Día laboral' ELSE 'Error' END; DBMS_OUTPUT.PUT_LINE('Hoy es ' || v_nombre_dia); DBMS_OUTPUT.PUT_LINE('Tipo: ' || v_tipo_dia); END; /

4. CONCEPTOS INTERMEDIOS

4.1 Cursores Básicos

INTERMEDIO

Definición

Los cursores son punteros a conjuntos de resultados que permiten procesar filas de una consulta una por una.

DECLARE CURSOR c_empleados IS SELECT id_emp, pnombre_emp, appaterno_emp, sueldo_base FROM empleado WHERE sueldo_base > 500000 ORDER BY sueldo_base DESC; v_emp_record c_empleados%ROWTYPE; BEGIN OPEN c_empleados; LOOP FETCH c_empleados INTO v_emp_record; EXIT WHEN c_empleados%NOTFOUND; DBMS_OUTPUT.PUT_LINE('ID: ' || v_emp_record.id_emp || ' - ' || v_emp_record.pnombre_emp || ' ' || v_emp_record.appaterno_emp || ' - $' || TO_CHAR(v_emp_record.sueldo_base, '999,999')); END LOOP; CLOSE c_empleados; DBMS_OUTPUT.PUT_LINE('Total procesado: ' || c_empleados%ROWCOUNT); END; /
Atributos de Cursor:
  • %FOUND: TRUE si la última operación FETCH encontró una fila
  • %NOTFOUND: TRUE si la última operación FETCH no encontró fila
  • %ROWCOUNT: Número de filas procesadas hasta el momento
  • %ISOPEN: TRUE si el cursor está abierto

4.2 Ciclos FOR con Cursores

INTERMEDIO

Cursor FOR Loop

Forma simplificada de procesar cursores que maneja automáticamente la apertura, cierre y fetch del cursor.

DECLARE v_total_sueldos NUMBER := 0; v_contador NUMBER := 0; BEGIN FOR emp_rec IN (SELECT pnombre_emp, appaterno_emp, sueldo_base FROM empleado WHERE sueldo_base BETWEEN 400000 AND 1000000 ORDER BY sueldo_base) LOOP v_contador := v_contador + 1; v_total_sueldos := v_total_sueldos + emp_rec.sueldo_base; DBMS_OUTPUT.PUT_LINE(v_contador || '. ' || INITCAP(emp_rec.pnombre_emp || ' ' || emp_rec.appaterno_emp) || ' - $' || TO_CHAR(emp_rec.sueldo_base, '999,999')); END LOOP; DBMS_OUTPUT.PUT_LINE('----------------------------------------'); DBMS_OUTPUT.PUT_LINE('Total empleados: ' || v_contador); DBMS_OUTPUT.PUT_LINE('Suma total: $' || TO_CHAR(v_total_sueldos, '999,999,999')); DBMS_OUTPUT.PUT_LINE('Promedio: $' || TO_CHAR(v_total_sueldos/v_contador, '999,999')); END; /

4.3 Ciclos WHILE

INTERMEDIO

Estructura WHILE LOOP

Ejecuta un bloque de código repetidamente mientras una condición sea verdadera.

DECLARE v_contador NUMBER := 1; v_suma NUMBER := 0; v_numero NUMBER; BEGIN DBMS_OUTPUT.PUT_LINE('Números pares del 1 al 20:'); WHILE v_contador <= 20 LOOP IF MOD(v_contador, 2) = 0 THEN v_suma := v_suma + v_contador; DBMS_OUTPUT.PUT_LINE('Número: ' || v_contador || ' | Suma acumulada: ' || v_suma); END IF; v_contador := v_contador + 1; END LOOP; DBMS_OUTPUT.PUT_LINE('Suma total de pares: ' || v_suma); END; /

4.4 Ciclos FOR Numéricos

INTERMEDIO

FOR Loop Numérico

Itera un número específico de veces con un contador automático.

DECLARE v_factorial NUMBER := 1; v_fibonacci_anterior NUMBER := 0; v_fibonacci_actual NUMBER := 1; v_fibonacci_siguiente NUMBER; BEGIN DBMS_OUTPUT.PUT_LINE('Tabla del 7:'); FOR i IN 1..10 LOOP DBMS_OUTPUT.PUT_LINE('7 x ' || i || ' = ' || (7 * i)); END LOOP; DBMS_OUTPUT.PUT_LINE(''); DBMS_OUTPUT.PUT_LINE('Factorial de 8:'); FOR i IN 1..8 LOOP v_factorial := v_factorial * i; DBMS_OUTPUT.PUT_LINE('!' || i || ' = ' || v_factorial); END LOOP; DBMS_OUTPUT.PUT_LINE(''); DBMS_OUTPUT.PUT_LINE('Secuencia Fibonacci (10 términos):'); DBMS_OUTPUT.PUT_LINE('F(0) = ' || v_fibonacci_anterior); DBMS_OUTPUT.PUT_LINE('F(1) = ' || v_fibonacci_actual); FOR i IN 2..9 LOOP v_fibonacci_siguiente := v_fibonacci_anterior + v_fibonacci_actual; DBMS_OUTPUT.PUT_LINE('F(' || i || ') = ' || v_fibonacci_siguiente); v_fibonacci_anterior := v_fibonacci_actual; v_fibonacci_actual := v_fibonacci_siguiente; END LOOP; END; /

4.5 Registros (Records)

INTERMEDIO

Definición

Los registros son estructuras de datos que agrupan elementos relacionados de diferentes tipos de datos.

DECLARE -- Registro usando %ROWTYPE v_empleado empleado%ROWTYPE; -- Registro personalizado TYPE tipo_resumen_empleado IS RECORD ( nombre_completo VARCHAR2(100), edad NUMBER, anos_servicio NUMBER, sueldo_categoria VARCHAR2(20), descuentos_totales NUMBER ); v_resumen tipo_resumen_empleado; BEGIN -- Usar registro %ROWTYPE SELECT * INTO v_empleado FROM empleado WHERE id_emp = 100; DBMS_OUTPUT.PUT_LINE('=== Datos Completos ==='); DBMS_OUTPUT.PUT_LINE('ID: ' || v_empleado.id_emp); DBMS_OUTPUT.PUT_LINE('Nombre: ' || v_empleado.pnombre_emp || ' ' || v_empleado.appaterno_emp); DBMS_OUTPUT.PUT_LINE('Sueldo: $' || TO_CHAR(v_empleado.sueldo_base, '999,999')); -- Llenar registro personalizado v_resumen.nombre_completo := INITCAP(v_empleado.pnombre_emp || ' ' || NVL(v_empleado.snombre_emp || ' ', '') || v_empleado.appaterno_emp || ' ' || v_empleado.apmaterno_emp); v_resumen.edad := FLOOR(MONTHS_BETWEEN(SYSDATE, v_empleado.fecha_nac) / 12); v_resumen.anos_servicio := FLOOR(MONTHS_BETWEEN(SYSDATE, v_empleado.fecha_contrato) / 12); v_resumen.sueldo_categoria := CASE WHEN v_empleado.sueldo_base > 1000000 THEN 'ALTO' WHEN v_empleado.sueldo_base > 500000 THEN 'MEDIO' ELSE 'BÁSICO' END; DBMS_OUTPUT.PUT_LINE(''); DBMS_OUTPUT.PUT_LINE('=== Resumen Procesado ==='); DBMS_OUTPUT.PUT_LINE('Nombre: ' || v_resumen.nombre_completo); DBMS_OUTPUT.PUT_LINE('Edad: ' || v_resumen.edad || ' años'); DBMS_OUTPUT.PUT_LINE('Antigüedad: ' || v_resumen.anos_servicio || ' años'); DBMS_OUTPUT.PUT_LINE('Categoría salarial: ' || v_resumen.sueldo_categoria); EXCEPTION WHEN NO_DATA_FOUND THEN DBMS_OUTPUT.PUT_LINE('Empleado no encontrado'); END; /

4.6 VARRAYs (Variable Arrays)

INTERMEDIO

Definición

Los VARRAYs son arrays de tamaño fijo que almacenan elementos del mismo tipo de datos en posiciones numeradas.
